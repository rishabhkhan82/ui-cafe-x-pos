  id: number;

  parent_id?: string; // null for root level
  path?: string;
  icon?: string;
  is_active: boolean;
  display_order: number;
  type?: 'PARENT' | 'ACTION'; // Menu type: parent container or action item (optional for backward compatibility)

  updated_by: number;
}
